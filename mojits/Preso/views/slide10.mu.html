Models

  • Models have access to configuration, autoloads, and other models.
  • Any additional data must be passed in through the controller.

YUI.add('ModelFoo', function(Y, NAME)
{
    Y.mojito.models[NAME] =
    {
        init: function(config) {
            this.config = config;
        },
        
        getData: function(args, cb) {
            // do something
        }
    };
}, '0.0.1', {requires: []});
